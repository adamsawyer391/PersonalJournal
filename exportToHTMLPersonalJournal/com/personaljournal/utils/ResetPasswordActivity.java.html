91       private void changePassword(String newPassword){
92           user.updatePassword(newPassword).addOnCompleteListener(new OnCompleteListener<Void>() {
93               @Override
94               public void onComplete(@NonNull Task<Void> task) {
95                   Toast toast = Toasty.success(mContext, "Your password is updated!", Toasty.LENGTH_LONG);
96                   toast.setGravity(Gravity.CENTER, 0, 0);
97                   toast.show();
98               }
99           });
100      }
101  
102      private void authenticateUser(){
103          String mEmail = etEmail.getText().toString();
104          String mPassword = etPassword.getText().toString();
105          if (checkInputs(mEmail, mPassword)){
106              AuthCredential authCredential = EmailAuthProvider.getCredential(mEmail, mPassword);
107              user.reauthenticate(authCredential).addOnCompleteListener(new OnCompleteListener<Void>() {
108                  @Override
109                  public void onComplete(@NonNull Task<Void> task) {
110                      if (task.isSuccessful()){
111                          authenticate_layout.setVisibility(View.GONE);
112                          reset_layout.setVisibility(View.VISIBLE);
113                      }else{
114                          Toast toast = Toasty.warning(mContext, "An error occurred. Please try again.", Toasty.LENGTH_LONG);
115                          toast.setGravity(Gravity.CENTER, 0, 0);
116                          toast.show();
117                      }
118                  }
119              });
120          }
121      }
122  
123      private boolean checkInputs(String email, String password){
124          if (TextUtils.isEmpty(email) || TextUtils.isEmpty(password)){
125              Toast toast = Toasty.warning(mContext, "Please fill out all fields", Toasty.LENGTH_LONG);
126              toast.setGravity(Gravity.CENTER, 0, 0);
127              toast.show();
128              return false;
129          }else if (!email.equals(mAuth.getCurrentUser().getEmail())){
130              Toast toast = Toasty.warning(mContext, "That is not the email associated with this account", Toasty.LENGTH_LONG);
131              toast.setGravity(Gravity.CENTER, 0, 0);
132              toast.show();
133              return false;
134          }
135          return true;
136      }
