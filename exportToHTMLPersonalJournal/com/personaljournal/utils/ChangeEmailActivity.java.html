34   public class ChangeEmailActivity extends AppCompatActivity {
35   
36       Context mContext;
37   
38       FirebaseAuth mAuth = FirebaseAuth.getInstance();
39       String currentUserID = mAuth.getCurrentUser().getUid();
40       DatabaseReference databaseReference = FirebaseDatabase.getInstance().getReference().child("users").child(currentUserID);
41   
42       RelativeLayout authenticate_layout, change_email_layout;
43       EditText etEmail, etConfirmEmail, etNewEmail, etConfirmNewEmail, etPassword;
44       Button button_email, button_authenticate;
45   
46       AdView adView;
47   
48       @Override
49       protected void onCreate(@Nullable Bundle savedInstanceState) {
50           super.onCreate(savedInstanceState);
51           setContentView(R.layout.activity_change_email);
52           mContext = ChangeEmailActivity.this;
53   
54           adView = findViewById(R.id.adBanner);
55           AdRequest adRequest = new AdRequest.Builder().build();
56           adView.loadAd(adRequest);
57   
58           authenticate_layout = findViewById(R.id.authenticate_email);
59           change_email_layout = findViewById(R.id.change_email);
60   
61           etEmail = findViewById(R.id.etEmail);
62           etConfirmEmail = findViewById(R.id.etConfirmEmail);
63           etPassword = findViewById(R.id.password);
64   
65           etNewEmail = findViewById(R.id.etNewEmail);
66           etConfirmNewEmail = findViewById(R.id.etConfirmNewEmail);
67   
68           button_authenticate = findViewById(R.id.authenticate);
69           button_authenticate.setOnClickListener(new View.OnClickListener() {
70               @Override
71               public void onClick(View v) {
72                   String mEmail = etEmail.getText().toString();
73                   String mConfirmEmail = etConfirmEmail.getText().toString();
74                   String mPassword = etPassword.getText().toString();
75                   if (checkInputs(mEmail, mConfirmEmail, mPassword)){
76                       AuthCredential authCredential = EmailAuthProvider.getCredential(mEmail, mPassword);
77                       mAuth.getCurrentUser().reauthenticate(authCredential).addOnCompleteListener(new OnCompleteListener<Void>() {
78                           @Override
79                           public void onComplete(@NonNull Task<Void> task) {
80                               if (task.isSuccessful()){
81                                   authenticate_layout.setVisibility(View.GONE);
82                                   change_email_layout.setVisibility(View.VISIBLE);
83                               }
84                           }
85                       });
86                   }
87               }
88           });
89   
90           button_email = findViewById(R.id.button_email);
91           button_email.setOnClickListener(new View.OnClickListener() {
92               @Override
93               public void onClick(View v) {
94                   final String mNewEmail = etNewEmail.getText().toString();
95                   String mConfirmNewEmail = etConfirmNewEmail.getText().toString();
96                   if (checkOtherInputs(mNewEmail, mConfirmNewEmail)){
97                       mAuth.fetchSignInMethodsForEmail(mNewEmail).addOnCompleteListener(new OnCompleteListener<SignInMethodQueryResult>() {
98                           @Override
99                           public void onComplete(@NonNull Task<SignInMethodQueryResult> task) {
100                              if (task.getResult().getSignInMethods().size() >= 1){
101                                  Toast toasty = Toasty.warning(mContext, "That email is already in use", Toasty.LENGTH_LONG);
102                                  toasty.setGravity(Gravity.CENTER, 0, 0);
103                                  toasty.show();
104                                  //Toasty.error(mContext, "That email is already in use", Toasty.LENGTH_LONG).show();
105                              }else{
106                                  changeEmail(mNewEmail);
107                              }
108                          }
109                      });
110  
111                  }
112  
113              }
114          });
115      }
116  
117      private boolean checkOtherInputs(String mNewEmail, String mConfirmNewEmail) {
118          if (TextUtils.isEmpty(mNewEmail) || TextUtils.isEmpty(mConfirmNewEmail)){
119              Toasty.warning(mContext, "Please fill out both fields", Toasty.LENGTH_LONG).show();
120              return false;
121          }else if (!mNewEmail.equals(mConfirmNewEmail)){
122              Toasty.warning(mContext, "Your email and confirm email do not match", Toasty.LENGTH_LONG).show();
123              return false;
124          }
125          return true;
126      }
127  
128      private boolean checkInputs(String mEmail, String mConfirmEmail, String mPassword){
129  
130          if (TextUtils.isEmpty(mEmail) || TextUtils.isEmpty(mConfirmEmail) || TextUtils.isEmpty(mPassword)){
131              Toasty.warning(mContext, "Please fill out all fields", Toasty.LENGTH_LONG).show();
132              return false;
133          }else if (!mEmail.equals(mConfirmEmail)){
134              Toasty.warning(mContext, "Your email and confirm email do not match", Toasty.LENGTH_LONG).show();
135              return false;
136          }
137          else if (!mEmail.equals(mAuth.getCurrentUser().getEmail())){
138              Toasty.warning(mContext, "That is not the email associated with this account", Toasty.LENGTH_LONG).show();
139              return false;
140          }
141  
142          return true;
143      }
144  
145      private void changeEmail(final String mNewEmail){
146          mAuth.getCurrentUser().updateEmail(mNewEmail).addOnCompleteListener(new OnCompleteListener<Void>() {
147              @Override
148              public void onComplete(@NonNull Task<Void> task) {
149                  if (task.isSuccessful()){
150                      HashMap hashMap = new HashMap();
151                      hashMap.put("email", mNewEmail);
152                      databaseReference.updateChildren(hashMap);
153                      Toast toast = Toasty.success(mContext, "Your email has been successfully updated. You will login in with the new email",
154                              Toasty.LENGTH_LONG);
155                      toast.setGravity(Gravity.CENTER, 0, 0);
156                      toast.show();
157                  }
158              }
159          });
160      }
