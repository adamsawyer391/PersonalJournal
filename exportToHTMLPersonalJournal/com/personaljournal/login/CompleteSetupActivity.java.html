60       private void checkInputs() {
61           if (TextUtils.isEmpty(fullname.getText().toString()) || TextUtils.isEmpty(location.getText().toString())){
62               Toasty.warning(mContext, "Please fill out all fields", Toasty.LENGTH_LONG).show();
63           }else{
64               FirebaseAuth mAuth = FirebaseAuth.getInstance();
65               String currentUserID = mAuth.getCurrentUser().getUid();
66               DatabaseReference databaseReference = FirebaseDatabase.getInstance().getReference();
67   
68               String mFullname, mLocation;
69               mFullname = fullname.getText().toString();
70               mLocation = location.getText().toString();
71   
72               HashMap hashMap = new HashMap();
73               hashMap.put("fullname", mFullname);
74               hashMap.put("location", mLocation);
75   
76               databaseReference.child("users").child(currentUserID).updateChildren(hashMap).addOnCompleteListener(new OnCompleteListener() {
77                   @Override
78                   public void onComplete(@NonNull Task task) {
79                       if (task.isSuccessful()){
80                           Intent intent = new Intent(mContext, MainActivity.class);
81                           intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK | Intent.FLAG_ACTIVITY_CLEAR_TASK);
82                           startActivity(intent);
83                           finish();
84                       }else{
85                           Toasty.error(mContext, "Something went wrong. Please try again", Toasty.LENGTH_LONG).show();
86                       }
87                   }
88               });
89           }
90       }
