286      private void saveAsPDF() {
287          Document document = new Document();
288          String fileName = UUID.randomUUID().toString();
289          String path = Environment.getExternalStorageDirectory()+"/"+fileName+".pdf";
290          try{
291              PdfWriter.getInstance(document, new FileOutputStream(path));
292              document.open();
293              String title = post_title.getText().toString();
294              String date = post_date.getText().toString();
295              String text = post_text.getText().toString();
296              document.add(new Paragraph(title));
297              document.add(new Paragraph(date));
298              document.add(new Paragraph(text));
299              Bitmap bitmap1, bitmap2, bitmap3, bitmap4;
300              if (image_one.getDrawable() != null){
301                  BitmapDrawable drawable = (BitmapDrawable) image_one.getDrawable();
302                  bitmap1 = drawable.getBitmap();
303                  bitmap1 = Bitmap.createScaledBitmap(bitmap1, 300, 200, false);
304                  ByteArrayOutputStream stream = new ByteArrayOutputStream();
305                  bitmap1.compress(Bitmap.CompressFormat.PNG, 100, stream);
306                  byte[] byteArray = stream.toByteArray();
307                  Image image = Image.getInstance(byteArray);
308                  document.add(image);
309              }
310              if (image_two.getDrawable() != null){
311                  BitmapDrawable drawable = (BitmapDrawable) image_two.getDrawable();
312                  bitmap2 = drawable.getBitmap();
313                  bitmap2 = Bitmap.createScaledBitmap(bitmap2, 300, 200, false);
314                  ByteArrayOutputStream stream = new ByteArrayOutputStream();
315                  bitmap2.compress(Bitmap.CompressFormat.JPEG, 100, stream);
316                  byte[] bytes = stream.toByteArray();
317                  Image image = Image.getInstance(bytes);
318                  document.add(image);
319              }
320              if (image_three.getDrawable() != null){
321                  BitmapDrawable drawable = (BitmapDrawable) image_three.getDrawable();
322                  bitmap3 = drawable.getBitmap();
323                  bitmap3 = Bitmap.createScaledBitmap(bitmap3, 300, 200, false);
324                  ByteArrayOutputStream stream = new ByteArrayOutputStream();
325                  bitmap3.compress(Bitmap.CompressFormat.JPEG, 100, stream);
326                  byte[] bytes = stream.toByteArray();
327                  Image image = Image.getInstance(bytes);
328                  document.add(image);
329              }
330              if (image_four.getDrawable() != null){
331                  BitmapDrawable drawable = (BitmapDrawable) image_four.getDrawable();
332                  bitmap4 = drawable.getBitmap();
333                  bitmap4 = Bitmap.createScaledBitmap(bitmap4, 300, 200, false);
334                  ByteArrayOutputStream stream = new ByteArrayOutputStream();
335                  bitmap4.compress(Bitmap.CompressFormat.JPEG, 100, stream);
336                  byte[] bytes = stream.toByteArray();
337                  Image image = Image.getInstance(bytes);
338                  document.add(image);
339              }
340              document.close();
341              emailDocument(path);
342          }catch (Exception e){
343              e.printStackTrace();
344          }

347      private void emailDocument(String document) {
348          Intent intent = new Intent(Intent.ACTION_SENDTO);
349          intent.putExtra(Intent.EXTRA_STREAM, Uri.parse("file://"+document));
350          intent.setData(Uri.parse("mailto:"));
351          intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
352          startActivity(intent);
353      }

360      private void loadData(){
361          databaseReference.child(mExtra).addValueEventListener(new ValueEventListener() {
362              @Override
363              public void onDataChange(@NonNull DataSnapshot dataSnapshot) {
364                  if (dataSnapshot.exists()){
365                      if (dataSnapshot.hasChild("title")){
366                          String title = dataSnapshot.child("title").getValue().toString();
367                          post_title.setText(title);
368                      }
369                      if (dataSnapshot.hasChild("text")){
370                          String text = dataSnapshot.child("text").getValue().toString();
371                          post_text.setText(text);
372                      }
373                      if (dataSnapshot.hasChild("time") && dataSnapshot.hasChild("day") && dataSnapshot.hasChild("date_created")){
374                          String time = dataSnapshot.child("time").getValue().toString();
375                          String day = dataSnapshot.child("day").getValue().toString();
376                          String date_created = dataSnapshot.child("date_created").getValue().toString();
377                          post_date.setText(date_created + " " + time + " | " + day);
378                      }
379                      //get design components
380                      if (dataSnapshot.hasChild("text_size")){
381                          String textSize = dataSnapshot.child("text_size").getValue().toString();
382                          post_text.setTextSize(Integer.parseInt(textSize));
383                      }
384                      if (dataSnapshot.hasChild("text_color")){
385                          String textColor = dataSnapshot.child("text_color").getValue().toString();
386                          post_text.setTextColor(Color.parseColor("#"+textColor));
387                      }
388                      if (dataSnapshot.hasChild("bg_color")){
389                          String bgColor = dataSnapshot.child("bg_color").getValue().toString();
390                          background.setBackgroundColor(Color.parseColor("#"+bgColor));
391                      }
392                      displayImages(dataSnapshot);
393  
394                  }else{
395                      Intent intent = new Intent(mContext, MainActivity.class);
396                      intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK | Intent.FLAG_ACTIVITY_CLEAR_TASK);
397                      startActivity(intent);
398                      finish();
399                  }
400              }
401  
402              @Override
403              public void onCancelled(@NonNull DatabaseError databaseError) {
404  
405              }
406          });
407      }
